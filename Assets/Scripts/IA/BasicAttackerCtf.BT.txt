tree("Root")
	repeat
		while isAliveCtf
			mute
				sequence
					checkForEnemies
					fallback
						tree "Priority"
						tree "FlagPickup"
						tree "Combat"
						tree "Help"
						tree "Objective"

tree "Priority"
	while hasFlag
		sequence
			GoBackToBase
			while hasEnemyInSight
				sequence
					CallForHelp
					ShootEnemy

tree "FlagPickup"
	while teamFlagPickable
		sequence
			GoToTeamFlag
			while hasEnemyInSight
				sequence
					CallForHelp
					ShootEnemy
	
tree "Combat"
	while hasEnemyInSight
		sequence
			fallback
				
				ShootEnemy
			FollowEnemy
	
	
tree "Help"
	while hasAllyInNeed
		sequence
			GoToHelp
			while hasEnemyInSight
				ShootEnemy
	
tree "Objective"
	fallback
		while enemyFlagInSight
			GoToEnemyFlag
		Explore
